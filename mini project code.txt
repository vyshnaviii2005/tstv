    let isSignup = true;
    let dataset = [];

    // Toggle Signup/Login
    function toggleForm() {
      const title = document.getElementById("formTitle");
      const button = document.querySelector(".form-box button");
      const toggleText = document.querySelector(".toggle");
      const message = document.getElementById("message");
      isSignup = !isSignup;
      message.textContent = "";
      if (isSignup) {
        title.textContent = "Create your account to get personalized fertilizer recommendations.";
        button.textContent = "Sign Up";
        toggleText.textContent = "Login here";
      } else {
        title.textContent = "Login to your account to access personalized recommendations.";
        button.textContent = "Login";
        toggleText.textContent = "Create one";
      }
    }

    // Handle Signup/Login
    function handleSubmit() {
      const email = document.getElementById("email").value.trim();
      const phone = document.getElementById("phone").value.trim();
      const message = document.getElementById("message");

      if (!email.endsWith("@gmail.com")) {
        message.style.color = "red";
        message.textContent = "Please enter a valid Gmail address.";
        return;
      }
      if (phone.length < 10 || isNaN(phone)) {
        message.style.color = "red";
        message.textContent = "Please enter a valid phone number.";
        return;
      }

      if (isSignup) {
        localStorage.setItem("userEmail", email);
        localStorage.setItem("userPhone", phone);
        message.style.color = "green";
        message.textContent = "Signup successful! You can now login.";
        document.getElementById("email").value = "";
        document.getElementById("phone").value = "";
      } else {
        const savedEmail = localStorage.getItem("userEmail");
        const savedPhone = localStorage.getItem("userPhone");
        if (email === savedEmail && phone === savedPhone) {
          message.style.color = "green";
          message.textContent = "Login successfull! Welcome To Your Smrt Fertilizer System";
          setTimeout(() => {
            document.getElementById("loginPage").classList.add("hidden");
            document.getElementById("mainPage").classList.remove("hidden");
          }, 1000);
        } else {
          message.style.color = "red";
          message.textContent = "Incorrect email or phone number.";
        }
      }
    }

    // Logout Function
    function logout() {
      localStorage.removeItem("chatData");
      document.getElementById("mainPage").classList.add("hidden");
      document.getElementById("loginPage").classList.remove("hidden");
    }

    // Fertilizer Recommendation
    function recommend() {
      const n = parseFloat(document.getElementById('n').value);
      const p = parseFloat(document.getElementById('p').value);
      const k = parseFloat(document.getElementById('k').value);
      const ph = parseFloat(document.getElementById('ph').value);
      const soil = document.getElementById('soil').value;
      const temp = document.getElementById('temp').value;
      const humidity = document.getElementById('humidity').value;

      if (ph < 0 || isNaN(ph)) {
        alert("Please enter a valid pH (0 or greater).");
        return;
      }

      let fertilizer = "DAP";
      if (n < 50) fertilizer = "Urea (Nitrogen-rich)";
      else if (p < 50) fertilizer = "Single Super Phosphate (SSP)";
      else if (k < 50) fertilizer = "Muriate of Potash (MOP)";
      else if (ph < 6.5) fertilizer = "Dolomite Lime";
      else if (soil === "black") fertilizer = "Ammonium Sulphate";
      else if (soil === "sandy") fertilizer = "Compost or Organic Manure";

      const resultText = `<b>Recommended Fertilizer:</b> ${fertilizer}<br>
        <b>Reason:</b> Based on soil type (${soil}), pH ${ph}, and nutrient balance.`;

      document.getElementById('result').innerHTML = resultText;

      dataset.push({ temp, humidity, ph, n, p, k, soil, fertilizer });
      localStorage.setItem("fertilizerData", JSON.stringify(dataset));
      document.getElementById('datasetInfo').innerText = `Saved ${dataset.length} entries in dataset.`;
    }

    // Chatbot Functionality (same multilingual logic)
    function sendMessage() {
      const input = document.getElementById('userInput');
      const msg = input.value.trim();
      const lang = document.getElementById('langSelect').value;
      if (!msg) return;
      appendMessage('user', msg);
      input.value = '';

      let reply = "";
      const lower = msg.toLowerCase();

      if (lower.includes("fertilizer") || msg.includes("‡§ñ‡§æ‡§¶")) {
        reply = getTranslation("The best fertilizer is Urea or DAP depending on your soil.", lang);
      } else if (lower.includes("ph")) {
        reply = getTranslation("If pH is below 6.5, use Dolomite Lime. For high pH, use Sulfur.", lang);
      } else if (lower.includes("rain") || msg.includes("‡§¨‡§æ‡§∞‡§ø‡§∂")) {
        reply = getTranslation("Reduce nitrogen fertilizers during heavy rainfall.", lang);
      } else if (lower.includes("soil")) {
        reply = getTranslation("Clay and loamy soils retain nutrients well, sandy soils need organic compost.", lang);
      } else if (lower.includes("thank")) {
        reply = getTranslation("You're welcome! Happy farming!", lang);
      } else if (lower.includes("hello") || msg.includes("‡§®‡§Æ‡§∏‡•ç‡§§‡•á")) {
        reply = getTranslation("Hello! How can I assist you with your crops today?", lang);
      } else {
        reply = getTranslation("I can guide you about fertilizers, soil, rainfall, and pH in your language!", lang);
      }

      appendMessage('bot', reply);
      speak(reply, lang);

      let chats = JSON.parse(localStorage.getItem("chatData")) || [];
      chats.push({ role: 'user', msg }, { role: 'bot', reply });
      localStorage.setItem("chatData", JSON.stringify(chats));
    }

    function appendMessage(sender, text) {
      const chatbot = document.getElementById('chatbot');
      const div = document.createElement('div');
      div.classList.add('message', sender);
      div.innerText = text;
      chatbot.appendChild(div);
      chatbot.scrollTop = chatbot.scrollHeight;
    }

    function startVoice() {
      const lang = document.getElementById('langSelect').value;
      const recognition = new(window.SpeechRecognition || window.webkitSpeechRecognition)();
      recognition.lang = lang;
      recognition.start();
      recognition.onresult = function(event) {
        const transcript = event.results[0][0].transcript;
        document.getElementById('userInput').value = transcript;
        sendMessage();
      }
    }

    function speak(text, lang) {
      const synth = window.speechSynthesis;
      const utter = new SpeechSynthesisUtterance(text);
      utter.lang = lang;
      synth.speak(utter);
    }
